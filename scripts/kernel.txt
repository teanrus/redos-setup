# ункция установки ядра
install_kernel() {
    if [ "$OS_MAJOR_VERSION" = "7" ]; then
        echo -e "${BLUE}становка ядра...${NC}"
        dnf install -y redos-kernels6-release
        check_success "становка ядра"
        update_grub
    else
        echo -e "${YELLOW}Skipping становка ядра for RED OS 8+${NC}"
    fi
}
